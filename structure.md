# Структура проекта

```
project/
├── config/                 # Конфигурационные файлы
│   ├── config.json         # Общий файл конфигурации для всего приложения
│   ├── bot_config.json     # Настройки бота
│   ├── app_config.json     # Настройки бота
│   └── config.py           # Настройки веб-приложения
├── data/                   # Директория для хранения файлов данных
│   └── database.db         # Файл базы данных SQLite
├── core/                   # Основные компоненты проекта
│   ├── database.py         # Настройки подключения к БД
│   ├── models.py           # Общие модели данных
│   └── utils.py            # Утилиты
├── services/
│   ├── bot/                # Телеграм бот
│   │   ├── handlers.py     # Обработчики команд
│   │   ├── keyboards.py    # Клавиатуры бота
│   │   ├── middleware.py   # Промежуточное ПО
│   │   └── utils.py        # Утилиты для бота
│   ├── web_app/            # Основное веб-приложение
│   │   ├── routes/         # Файлы маршрутов
│   │   ├── templates/      # Шаблоны
│   │   ├── static/         # Статические файлы
│   │   ├── auth/           # Аутентификация
│   │   └── utils.py        # Утилиты для веб-приложения
│   └── api/                # REST API сервис
│       ├── endpoints/      # Точки API
│       │   ├── bot.py      # Эндпоинт для бота
│       │   ├── app.py      # Эндпоинты для веб-приложения
│       │   └── health.py   # Эндпоинт для проверки доступности
│       ├── serializers.py # Сериализаторы данных
│       ├── validators.py   # Валидаторы
│       └── utils.py        # Утилиты для API
├── mcp/                    # Модуль для MCP сервера
│   ├── server.py           # Основной сервер
│   ├── protocols/          # Протоколы обмена
│   └── adapters/           # Адаптеры для разных сервисов
├── tests/                  # Тесты для всех компонентов
│   ├── unit/               # Модульные тесты
│   ├── integration/        # Интеграционные тесты
│   └── e2e/                # Сквозные тесты
├── docs/                   # Документация проекта
│   ├── architecture.md     # Архитектура
│   ├── api.md              # API документация
│   └── deployment.md       # Инструкции по развертыванию
├── scripts/                # Скрипты для запуска, деплоя и обслуживания
│   ├── deploy.sh           # Скрипт деплоя
│   └── backup.sh           # Скрипт бэкапа
├── main.py                 # Основной файл запуска приложения
├── .env                    # Файл переменных окружения
├── README.md               # Основная документация проекта
└── Dockerfile              # Один Dockerfile для всего приложения
